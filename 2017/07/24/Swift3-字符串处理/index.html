Swift3-字符串处理

Veröffentlicht am 2017-07-24

###以下是一些字符串的处理：

###字符串的定义

var str1 = "hello" //字符串变量

let str2 = "hello" //字符串常量

let str3 = "" //空字符串

let str4 = String() //空字符串

###字符串的连接
//使用 + 号

let str1 = "hello"

let str2 = "swift"

let str3 = str1 + str2

print(str3)

//使用字符串插值(该方法也可用于字符串与其他类型值的连接)

let str4 = "\(str1)\(str2)"
print(str4)

###字符串枚举字符

let str = "字符串枚举字符"

for ch in str.characters {

print(ch)

}

###字符串的长度

let str11 = "hello word"
print(str11.characters.count)

###首字母大写capitalized

let str111 = "hello"
let str222 = str111.capitalized

print(str222)

###字符串全部转为小写lowrecase

let str111 = "HELLO"
let str22 = str111.lowercased()

print(str22)

###按照下标截取子字符串

let str1 = "HELLO word"

//从头开始截取
let str2 = str1.substring(from: str1.startIndex)

//截取到尾部
let str3 = str1.substring(to: str1.endIndex)

//获取截取范围
let i = str1.index(str1.startIndex, offsetBy: 3)

let j = str1.index(str1.startIndex, offsetBy: 9)

let str4 = str1.substring(with: i..<j)

print(str2)

print(str3)

print(str4)

###字符串的比较 判断两个字符串是否相等

let str22 = "hello"

let str222 = "hello3"

if str22 == str222 {

print("两个字符串相等")

} else {

print("两个字符串不相等")

}

###判断字符串是否包含子串

let str33 = "hello swift"

let str3311 = "SWIFT"

let range11 = str33.range(of: str3311)//正向查找

let range22 = str33.range(of: str3311, options: .backwards)//反向查找

let range33 = str33.range(of:str3311, options: .caseInsensitive, range:nil , locale:nil)//忽略大小写

if range11 != nil {

print("str33中包含着子串")

} else {

print("str33中不包含子串")

}

###字符串的分解

var str1 = "hello,word,hello,you,!"

let strArray1:[String] = str1.components(separatedBy: ",")

for index in strArray1 {

print(index)

}

###字符串替换replace

var str1 = "hello word,hello you!"

let str2 = str1.replacingOccurrences(of: "word", with: "swift")//字符串替换字符串

let i = str1.index(str1.startIndex, offsetBy: 6)

let j = str1.index(str1.startIndex, offsetBy: 12)

let str3 = str1.replacingCharacters(in: (i..<j), with: "swift")
//字符串替换某个范围的字符串


let str4 = str1.replacingOccurrences(of: "WorD", with: "SWIFT", options: .caseInsensitive, range: nil)//忽略大小写替换

print(str2)

print(str3)

print(str4)
